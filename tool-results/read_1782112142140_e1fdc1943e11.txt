   423тЖТ   423тЖТ  // Flexible header mapping тАФ recognizes common control-catalog column names
   424тЖТ   424тЖТ  const HEADER_ALIASES: Record<string, string[]> = {
   425тЖТ   425тЖТ    ref:        ['ref', 'control id', 'criteria id', 'citation', 'control ref', 'id', 'reference', 'clause', 'article'],

   432тЖТ   432тЖТ  function normalizeHeader(h: string): string {
   433тЖТ   433тЖТ    const lower = h.trim().toLowerCase()
   434тЖТ   434тЖТ    for (const [canonical, aliases] of Object.entries(HEADER_ALIASES)) {
   435тЖТ   435тЖТ      if (aliases.includes(lower)) return canonical

   440тЖТ   440тЖТ  // Map a raw row object (with various possible header names) to the {ref, title, ...} format
   441тЖТ   441тЖТ  function mapRow(rawRow: Record<string, string>): any {
   442тЖТ   442тЖТ    const mapped: any = {}
   443тЖТ   443тЖТ    for (const [rawKey, val] of Object.entries(rawRow)) {
   444тЖТ   444тЖТ      const canonical = normalizeHeader(rawKey)
   445тЖТ   445тЖТ      if (canonical && val) {
   446тЖТ   446тЖТ        if (!mapped[canonical]) mapped[canonical] = val.trim()
   447тЖТ   447тЖТ      }
   448тЖТ   448тЖТ    }
   449тЖТ   449тЖТ    return mapped
   450тЖТ   450тЖТ  }

   452тЖТ   452тЖТ  function parseCsv(text: string): any[] {
   453тЖТ   453тЖТ    const lines = text.trim().split(/\r?\n/).filter(Boolean)
   454тЖТ   454тЖТ    if (lines.length < 2) return []
   455тЖТ   455тЖТ    const headers = lines[0].split(',').map((h) => h.trim().replace(/^"|"$/g, ''))
   456тЖТ   456тЖТ    return lines.slice(1).map((line) => {
   457тЖТ   457тЖТ      // CSV parse that handles quoted fields with commas
   458тЖТ   458тЖТ      const vals: string[] = []
   459тЖТ   459тЖТ      let cur = ''
   460тЖТ   460тЖТ      let inQuotes = false
   461тЖТ   461тЖТ      for (let i = 0; i < line.length; i++) {
   462тЖТ   462тЖТ        const ch = line[i]
   463тЖТ   463тЖТ        if (ch === '"') { inQuotes = !inQuotes; continue }
   464тЖТ   464тЖТ        if (ch === ',' && !inQuotes) { vals.push(cur); cur = ''; continue }
   465тЖТ   465тЖТ        cur += ch
   466тЖТ   466тЖТ      }
   467тЖТ   467тЖТ      vals.push(cur)
   468тЖТ   468тЖТ      const obj: Record<string, string> = {}
   469тЖТ   469тЖТ      headers.forEach((h, i) => { obj[h] = (vals[i] || '').trim().replace(/^"|"$/g, '') })
   470тЖТ   470тЖТ      return mapRow(obj)
   471тЖТ   471тЖТ    })
   472тЖТ   472тЖТ  }
   473тЖТ   473тЖТ
   474тЖТ   474тЖТ  // Parse an HTML table (from .docx via mammoth) into control rows
   475тЖТ   475тЖТ  function parseHtmlTable(html: string): any[] {
   476тЖТ   476тЖТ    const rows: any[] = []
   477тЖТ   477тЖТ    // Use DOMParser to extract table rows
   478тЖТ   478тЖТ    if (typeof DOMParser === 'undefined') return rows
   479тЖТ   479тЖТ    const doc = new DOMParser().parseFromString(html, 'text/html')
   480тЖТ   480тЖТ    const tables = doc.querySelectorAll('table')
   481тЖТ   481тЖТ    tables.forEach((table) => {
   482тЖТ   482тЖТ      const trs = table.querySelectorAll('tr')
   483тЖТ   483тЖТ      if (trs.length < 2) return
   484тЖТ   484тЖТ      // First row = headers
   485тЖТ   485тЖТ      const headerCells = Array.from(trs[0].querySelectorAll('td, th')).map((c) => c.textContent?.trim() || '')
   486тЖТ   486тЖТ      if (headerCells.length === 0) return
   487тЖТ   487тЖТ      // Data rows
   488тЖТ   488тЖТ      for (let i = 1; i < trs.length; i++) {
   489тЖТ   489тЖТ        const cells = Array.from(trs[i].querySelectorAll('td, th')).map((c) => c.textContent?.trim() || '')
   490тЖТ   490тЖТ        if (cells.length === 0) continue
   491тЖТ   491тЖТ        const obj: Record<string, string> = {}
   492тЖТ   492тЖТ        headerCells.forEach((h, j) => { if (h) obj[h] = cells[j] || '' })
   493тЖТ   493тЖТ        rows.push(mapRow(obj))
   494тЖТ   494тЖТ      }
   495тЖТ   495тЖТ    })
   496тЖТ   496тЖТ    return rows
   497тЖТ   497тЖТ  }

   499тЖТ   499тЖТ  function parseTextToControls(text: string): any[] {
   500тЖТ   500тЖТ    const trimmed = text.trim()
   501тЖТ   501тЖТ    if (!trimmed) return []
   502тЖТ   502тЖТ    if (trimmed.startsWith('[')) {
   503тЖТ   503тЖТ      // JSON array тАФ map keys too
   504тЖТ   504тЖТ      const arr = JSON.parse(trimmed)
   505тЖТ   505тЖТ      return Array.isArray(arr) ? arr.map((r: any) => {
   506тЖТ   506тЖТ        // Normalize keys
   507тЖТ   507тЖТ        const obj: Record<string, string> = {}
   508тЖТ   508тЖТ        for (const [k, v] of Object.entries(r)) { obj[k] = String(v ?? '') }
   509тЖТ   509тЖТ        return mapRow(obj)
   510тЖТ   510тЖТ      }) : []
   511тЖТ   511тЖТ    }
   512тЖТ   512тЖТ    return parseCsv(trimmed)
   513тЖТ   513тЖТ  }
